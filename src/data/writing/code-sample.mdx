### Python

```python
def fib():
    a, b = 0, 1
    while True:            # First iteration:
        yield a            # yield 0 to start with and then
        a, b = b, a + b    # a will now be 1, and b will also be 1, (0 + 1)

for index, fibonacci_number in zip(range(10), fib()):
     print('{i:3}: {f:3}'.format(i=index, f=fibonacci_number))
```

### Docker

```docker
FROM haproxy:2.3.5-alpine

# declare configs envs
ENV HAPROXY_USER=haproxy

# add the user with permissions
RUN mkdir --parents /var/lib/$HAPROXY_USER && \
  chown -R $HAPROXY_USER:$HAPROXY_USER /var/lib/$HAPROXY_USER

RUN mkdir --parents /run/haproxy && \
    chown -R $HAPROXY_USER:$HAPROXY_USER /run/haproxy

# copy configs
COPY haproxy.cfg /usr/local/etc/haproxy/haproxy.cfg

# start with the specified configs
CMD ["haproxy", "-db", "-f", "/usr/local/etc/haproxy/haproxy.cfg"]

```

### NGINX

```nginx
upstream mk-upstream {
    server mk-upstream:9000;
}

server {
    listen 80 proxy_protocol;
    listen [::]:80 proxy_protocol;
    server_name example.com;

    set_real_ip_from example-proxy;
    real_ip_header proxy_protocol;

    # THREADING
    aio threads;
    directio 5m;
    
    # Locations

    # Block access to these locations
    location ~* /\.(?!well-known\/) {
        deny all;
    }

    location ~* (?:#.*#|\.(?:bak|conf|dist|fla|in[ci]|log|orig|psd|sh|sql|sw[op])|~)$ {
        deny all;
    }

    # Public locations

    location ~* \.(?:css|js|jpg|jpe?g|gif|png|svg)$ {
        # default headers
        include /etc/nginx/headers.conf;
        include /etc/nginx/cache.conf;
        expires 1y;
        proxy_pass http://mk-upstream;
    }

    location ~* \.(?:woff|woff2?)$ {
        # default headers
        include /etc/nginx/headers.conf;
        add_header Cache-Control 'public, max-age=31536000, immutable';
        add_header Access-Control-Allow-Origin '*';
        proxy_pass http://mk-upstream;
    }

    location / {
        # default headers
        include /etc/nginx/headers.conf;
        include /etc/nginx/cache.conf;
        expires 1d;
        proxy_pass http://mk-upstream;
    }
}
```
